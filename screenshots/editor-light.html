use std::fmt;
use serde::{Deserialize, Serialize};
/// A single color token with contrast metadata.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Token {
pub name: String,
pub hex: String,
pub contrast: f64,
pub style: FontStyle,
}
#[derive(Debug, Clone, Copy, PartialEq, Serialize, Deserialize)]
pub enum FontStyle {
Normal,
Bold,
Italic,
}
const AAA_MIN: f64 = 7.0;
const AA_MIN: f64 = 4.5;
impl Token {
/// Check if the token meets the required contrast for a variant.
pub fn meets_contrast(&self, variant: Variant) -> bool {
let min = match variant {
Variant::Dark => AAA_MIN,
Variant::Light => AA_MIN,
};
self.contrast >= min
}
pub fn luminance(&self) -> f64 {
let rgb = hex_to_rgb(&self.hex);
0.2126 * srgb_to_linear(rgb.0)
+ 0.7152 * srgb_to_linear(rgb.1)
+ 0.0722 * srgb_to_linear(rgb.2)
}
