import { useState, useEffect } from "react";
import { Theme, ColorToken } from "./types";
import { loadPalette } from "./lib/palette";
// WCAG AAA contrast ratios for every token
const MIN_CONTRAST = 7.0;
const WARM_HUE_RANGE: readonly [number, number] = [13, 40];
interface ThemeProviderProps {
children: React.ReactNode;
variant?: "dark" | "light";
fallback?: Theme;
}
export function ThemeProvider({ children, variant = "dark", fallback }: ThemeProviderProps) {
const [theme, setTheme] = useState<Theme | null>(fallback ?? null);
const [error, setError] = useState<string | null>(null);
useEffect(() => {
const palette = loadPalette(variant);
// Validate all tokens meet contrast minimum
for (const [name, token] of Object.entries(palette.tokens)) {
const ratio = contrastRatio(token.hex, palette.bg);
if (ratio < MIN_CONTRAST) {
setError(`Token ${name} fails AAA: ${ratio.toFixed(1)}:1`);
return;
}
}
setTheme(palette);
}, [variant, fallback]);
if (error) return <div className="theme-error">{error}</div>;
if (!theme) return <div className="loading" />;
return <ThemeContext.Provider value={theme}>{children}</ThemeContext.Provider>;
}
